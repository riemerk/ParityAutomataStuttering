\begin{theorem}
 Let $\phi$ be a $\mu$-TL formula. Without loss of generality assume assume $\phi$ is guarded and simple. There exists an equivalent NPA $\A$ where
 \[
  \Mod_\Psf(\phi) = \LL(\A)
 \]
\end{theorem}
